no imprimir aquellos que no tienen nada en stock

	"#{coffee} has #{attrs.in_stock}"
to_print.join " , "

in js

var attrs, coffee, to_print;

to_print = (function(){
	var _results;
	_results= [];

	for (coffee in coffees){
		attrs = coffees[coffee];
		if (attrs.in_stock > 0) _results.push("" + coffee + " has " + attrs.in_stock);
	}
	return _results;
})();
